ImageList Field

An image list field is a control that will display a set of images and some text.


            <af-form 
              [formDefinition]="myFormDefinition" 
              [(formData)]="myFormData" 
            </af-form>
        
        
          import {{'{'}} Component, OnInit {{'}'}} from '@angular/core';
    
          @Component({{'{'}}
            selector: 'app-image-list-field-example',
            templateUrl: './image-list-field-example.component.html',
            styleUrls: ['./image-list-field-example.component.css']
          })
          export class ImageListFieldExampleComponent {{'{'}}
          
            myFormData = {{'{'}}
              imageListField: 'value1'
            {{'}'}};
          
            myFormDefinition = {{'{'}}
              layout: [['imageListField']],
              fields: [
                {{'{'}}
                  name: 'imageListField',
                  fieldType: 'ImageList',
                  label: 'Image List (Single Select)',

                    {{'{'}} value: 'value1', imageUrl: 'https://via.placeholder.com/150x150' {{'}'}},

Multi Select

In some cases, you may require the user to have the ability to select more than one option. You can configure this by setting the 'isMultiple' flag on the image select field.


            <af-form 
              [formDefinition]="myFormDefinition" 
              [(formData)]="myFormData" 
            </af-form>
        
        
          import {{'{'}} Component, OnInit {{'}'}} from '@angular/core';
    
          @Component({{'{'}}
            selector: 'app-image-list-field-example',
            templateUrl: './image-list-field-example.component.html',
            styleUrls: ['./image-list-field-example.component.css']
          })
          export class ImageListFieldExampleComponent {{'{'}}
          
          myFormDataMultiSelect = {{'{'}}
              imageListField: ['value2', 'value3']
            {{'}'}};
          
            myFormDefinitionMultiSelect = {{'{'}}
              layout: ['imageListField']],
              fields: [
                {{'{'}}
                  name: 'imageListField',
                  fieldType: 'ImageList',
                  label: 'Image List (Multi-Select)',
                  isMultiple: true,
                  imageOptions: [
                    {{'{'}} value: 'value1', imageUrl: 'https://via.placeholder.com/150x150' {{'}'}},
                    {{'{'}} value: 'value2', imageUrl: 'https://via.placeholder.com/150x150' {{'}'}},
                    {{'{'}} value: 'value3', imageUrl: 'https://via.placeholder.com/150x150' {{'}'}},
                    {{'{'}} value: 'value4', imageUrl: 'https://via.placeholder.com/150x150' {{'}'}}
                  ]
                {{'}'}}
              ]
            {{'}'}};

Validations

            <af-form 
              [formDefinition]="myFormDefinition" 
              [(formData)]="myFormData" 
            </af-form>
        
        
            import {{'{'}} Component, OnInit {{'}'}} from '@angular/core';
    
            @Component({{'{'}}
              selector: 'app-image-list-field-example',
              templateUrl: './image-list-field-example.component.html',
              styleUrls: ['./image-list-field-example.component.css']
            })
            export class ImageListFieldExampleComponent {{'{'}}
            
                myFormDataValidation = {{'{'}}
                  imageListField: ''
                {{'}'}};
