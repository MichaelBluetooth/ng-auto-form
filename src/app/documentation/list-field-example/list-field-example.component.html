List Field

Lists are best used when the user must enter a value from a constrained set of possible values.


            <af-form 
              [formDefinition]="myFormDefinition" 
              [(formData)]="myFormData" 
            </af-form>
        
        
          import {{'{'}} Component {{'}'}} from '@angular/core';
    
          @Component({{'{'}}
            selector: 'list-example',
            templateUrl: './list-field-example.component.html'
          })
          export class ListExampleComponent {{'{'}}
          
            myFormData = {{'{'}}
              listField: ['Option 1', 'Option 2'],
            {{'}'}};
          
            myFormDefinition = {{'{'}}
              layout: [['listField1']],
              fields: [
                {{'{'}}
                  name: 'listField',
                  fieldType: 'MultiList',
                  label: 'List Field Label',
                  listOptions: ['Option 1', 'Option 2', 'Option 3']
                {{'}'}},
              ]
            {{'}'}}
          {{'}'}}

List Options As Objects

Sometimes the list options require the value be different than whats displayed. In these cases a list of object with "name" and "id" fields can be supplied. The "name" will be used for display and the "id" will be used as the selected value.


            <af-form 
              [formDefinition]="myFormDefinitionWithNameAndIds" 
              [(formData)]="myFormDataWithNameAndIds" 
            </af-form>
        
        
          import {{'{'}} Component, OnInit {{'}'}} from '@angular/core';
    
          @Component({{'{'}}
            selector: 'app-list-field-example',
            templateUrl: './list-field-example.component.html',
            styleUrls: ['./list-field-example.component.css']
          {{'}'}})
          export class ListFieldExampleComponent {{'{'}}
          
            myFormDataWithNameAndIds = {{'{'}}
              listField: 2
            {{'}'}};
          
            myFormDefinitionWithNameAndIds = {{'{'}}
              layout: [['listField']],
              fields: [
                {{'{'}}
                  name: 'listField',
                  fieldType: 'List',
                  label: 'List Label',
                  listOptions: [{{'{'}} id: 1, name: 'Option 1' }, {{'{'}} id: 2, name: 'Option 2' }]
                {{'}'}}
              ]
            {{'}'}};
          {{'}'}}

List Options As Custom Objects

In the even your list options is not a collection of strings or a collection of name/ids, you can provide your own field names to be used as the value and display.


            <af-form 
              [formDefinition]="myFormDefinitionWithCustomFields" 
              [(formData)]="myFormDataWithCustomFields" 
            </af-form>
        
        
          import {{'{'}} Component, OnInit {{'}'}} from '@angular/core';
    
          @Component({{'{'}}
            selector: 'app-list-field-example',
            templateUrl: './list-field-example.component.html',
            styleUrls: ['./list-field-example.component.css']
          {{'}'}})
          export class ListFieldExampleComponent {{'{'}}
          
            myFormDataWithCustomFields = {{'{'}}
              listField: 2
            {{'}'}};
          
            myFormDefinitionWithCustomFields = {{'{'}}
              layout: [['listField']],
              fields: [
                {{'{'}}
                  name: 'listField',
                  fieldType: 'List',
                  label: 'List Label',
                  displayFieldName: 'nodeId',
                  valueFieldName: 'nodeName',
                  listOptions: [{{'{'}} nodeId: 1, nodeName: 'Option 1' }, {{'{'}} nodeId: 2, nodeName: 'Option 2' {{'}'}}]
                {{'}'}}
              ]
            {{'}'}};
          {{'}'}}

Validation

Validators can be automatically applied to list fields.


            <af-form 
              [formDefinition]="myFormDefinitionWithCustomFields" 
              [(formData)]="myFormDataWithCustomFields" 
            </af-form>
        
        
          import {{'{'}} Component, OnInit {{'}'}} from '@angular/core';
    
          @Component({{'{'}}
            selector: 'app-list-field-example',
            templateUrl: './list-field-example.component.html',
            styleUrls: ['./list-field-example.component.css']
          {{'}'}})
          export class ListFieldExampleComponent {{'{'}}
          
            myFormDataWithRequiredFields = {{'{'}}
              listField: ''
            {{'}'}};
          
            myFormDefinitionWithRequiredFields = {{'{'}}
              layout: [['listField']],
              fields: [
                {{'{'}}
                  name: 'listField',
                  fieldType: 'List',
                  label: 'List Label',
                  listOptions: ['', 'Option 1', 'Option 2'],
                  validations: [{{'{'}} name: 'Required' {{'}'}}]
                {{'}'}}
              ]
            {{'}'}};    
          {{'}'}}
